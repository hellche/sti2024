(c) Function for detecting Slavic surname endings (suffix)

is_slavic_surname <- function(surname, specific_names) {
  # Define a pattern that captures common Slavic surname endings and any Cyrillic characters
  pattern_suffix <- "(?i).*(nin|nina|kin|kina|lyi|ov|ova|ev|eva|ski|ska|sky|skaya)$"
  pattern_cyrillic <- "[А-Яа-яЁё]"  # Matches any Cyrillic character
  
  # Check for Slavic suffix or Cyrillic characters
  has_slavic_suffix <- str_detect(surname, pattern_suffix)
  contains_cyrillic <- str_detect(surname, pattern_cyrillic)
  
  # Check if the surname is in the specific list
  is_in_specific_list <- surname %in% specific_names
  
  return(has_slavic_suffix | contains_cyrillic | is_in_specific_list)
}
